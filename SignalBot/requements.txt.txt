    const tg = window.Telegram.WebApp;
    
    // --- SIGNAL GENERATORLARI ---

    // Array ichidan tasodifiy noyob pozitsiyalarni tanlash
    function getRandomUniquePositions(max, count) {
        const positions = [];
        while (positions.length < count) {
            const pos = Math.floor(Math.random() * max);
            if (!positions.includes(pos)) {
                positions.push(pos);
            }
        }
        return positions;
    }

    function generateMinesSignalHTML() {
        const GRID_SIZE = 5;
        const NUM_SAFE = 3; 
        const safe_spots = getRandomUniquePositions(GRID_SIZE * GRID_SIZE, NUM_SAFE);
        
        let gridHTML = '<div class="mines-grid-display">';
        for (let i = 0; i < GRID_SIZE * GRID_SIZE; i++) {
            const isSafe = safe_spots.includes(i);
            const content = isSafe ? '‚úÖ' : '‚ùå'; 
            // Tailwind yordamida chizish
            const bgColor = isSafe ? 'bg-green-500' : 'bg-red-500';
            const shadow = isSafe ? 'shadow-lg shadow-green-700/50' : 'shadow-lg shadow-red-700/50';

            gridHTML += `<span class="${bgColor} text-white ${shadow}">${content}</span>`;
        }
        gridHTML += '</div>';

        const message = `
            üí£ <b>MINES UCHUN TASODIFIY SIGNAL</b> üí£
            <br><span class="text-xs text-gray-500">(25x25 maydon, 3 ta xavfsiz ‚úÖ)</span>
            ${gridHTML}
            <br>
            <div class="flex justify-around text-sm font-semibold">
                <span class="text-green-600">‚úÖ Xavfsiz</span>
                <span class="text-red-600">‚ùå Xavfli</span>
            </div>
        `;
        return message;
    }

    function generateGemsSignalHTML() {
        const GRID_SIZE = 3;
        const safe_spot = Math.floor(Math.random() * GRID_SIZE); // 0, 1, yoki 2
        
        let displayHTML = '<div class="flex justify-center space-x-3 mt-4">';
        for (let i = 0; i < GRID_SIZE; i++) {
            const isSafe = (i === safe_spot);
            const content = isSafe ? '‚úÖ' : 'üü¶';
            const bgColor = isSafe ? 'bg-green-500' : 'bg-blue-500';
            const shadow = isSafe ? 'shadow-lg shadow-green-700/50' : 'shadow-lg shadow-blue-700/50';

            displayHTML += `<span class="text-3xl p-3 rounded-xl ${bgColor} text-white ${shadow}">${content}</span>`;
        }
        displayHTML += '</div>';
        
        const message = `
            üíé <b>GEMS UCHUN TASODIFIY SIGNAL</b> üíé
            <br><span class="text-xs text-gray-500">(3 ta katakdan 1 tasi xavfsiz ‚úÖ)</span>
            ${displayHTML}
            <br>
            <div class="mt-3 text-base font-semibold">
                ‚úÖ Yashil ptichka aniqlandi!
            </div>
        `;
        return message;
    }

    function generateRouletteSignalHTML() {
        const colors = ['üî¥ Qizil', '‚ö´Ô∏è Qora'];
        const weights = [0.55, 0.45]; // Tikish uchun biroz yuqori ehtimollik berishimiz mumkin
        
        // Tasodifiy rangni tanlash (eng oddiy yo'li)
        const recommendedColorIndex = Math.random() < 0.5 ? 0 : 1;
        const recommendedColor = colors[recommendedColorIndex];
        
        // Yana bir ehtimollik signali (masalan, 80% hollarda to'g'ri)
        const isStrongSignal = Math.random() < 0.8; 
        
        let colorDisplayHTML = `<span class="roulette-color ${recommendedColorIndex === 0 ? 'bg-red-600' : 'bg-black'}">
                                    ${recommendedColor}
                                </span>`;

        const message = `
            üî¥ <b>ROULETTE UCHUN SIGNAL</b> ‚ö´Ô∏è
            <br><span class="text-xs text-gray-500">(Qizil / Qora)</span>
            <div class="roulette-display flex justify-center items-center space-x-2">
                TIKISH: ${colorDisplayHTML}
            </div>
            <div class="mt-4 text-sm font-semibold text-gray-700">
                <span class="${isStrongSignal ? 'text-green-600' : 'text-yellow-600'}">
                    ${isStrongSignal ? 'Kafolat: 80% muvaffaqiyat ehtimoli.' : 'Ehtiyot bo\'ling: oddiy signal.'}
                </span>
            </div>
        `;
        return message;
    }

    // --- ASOSIY FUNKSIYA ---

    function generateSignal(type) {
        tg.HapticFeedback.impactOccurred('light');
        const resultArea = document.getElementById('result-area');
        let signalHTML = '';
        let title = '';

        // Natija maydonini tozalash va yuklanish effektini berish
        resultArea.innerHTML = '<h3>Signal yaratilmoqda... üïí</h3>';

        // 500ms dan keyin signalni ko'rsatish (UX uchun)
        setTimeout(() => {
            switch(type) {
                case 'mines':
                    title = "üí£ Mines Signali";
                    signalHTML = generateMinesSignalHTML();
                    break;
                case 'gems':
                    title = "üíé Gems Signali";
                    signalHTML = generateGemsSignalHTML();
                    break;
                case 'roulette':
                    title = "üî¥ Roulette Signali";
                    signalHTML = generateRouletteSignalHTML();
                    break;
                default:
                    signalHTML = '<h3>Noma\'lum signal turi.</h3>';
            }

            // Natijani DOMga joylashtirish
            resultArea.innerHTML = signalHTML;
            
            // Telegram Main Button'ni yangilash (agar kerak bo'lsa)
            // Hozirda faqat UXni yaxshilaymiz
            tg.HapticFeedback.notificationOccurred('success');

        }, 500);
    }
